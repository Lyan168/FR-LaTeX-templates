\begin{frame}
    \frametitle{Les différentes alertes}
    \begin{itemize}
        \item Alerte standard: \stdalert{texte important}
        \item Alerte standard au deuxième clic: \stdalert[2]{texte important}
        \item Alerte standard au troisième clic: \stdalert[3]{texte important}
        \item Trois alertes pour les nombres:
              \begin{itemize}
                  \item Nombre positif: \galert{+5}
                  \item Nombre négatif: \ralert{-10}
                  \item Nombre nul: \balert{0.0}
              \end{itemize}
        \item Couleur seulement après le premier clic:
              \begin{itemize}
                  \item Nombre positif: \galert[2-3]{+5}
                  \item Nombre négatif: \ralert[2-3]{-10}
                  \item Nombre nul: \balert[2-3]{0.0}
              \end{itemize}
    \end{itemize}
\end{frame}
